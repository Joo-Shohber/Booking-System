## Lua Scripts — src/utils/lua-scripts.ts

Export as typed string constants.

BOOK_SLOT_SCRIPT:
  KEYS[1] = slot:{slotId}:available
  local current = redis.call("GET", KEYS[1])
  if current == false then return -2 end
  if tonumber(current) <= 0 then return -1 end
  return redis.call("DECR", KEYS[1])

RELEASE_SLOT_SCRIPT:
  KEYS[1] = slot:{slotId}:available
  return redis.call("INCR", KEYS[1])

RELEASE_LOCK_SCRIPT:
  if redis.call("get",KEYS[1])==ARGV[1] then return redis.call("del",KEYS[1]) else return 0 end

RATE_LIMIT_SCRIPT:
  KEYS[1] = rate limit key
  ARGV[1]=now(ms string) ARGV[2]=windowStart(ms string) ARGV[3]=ttl(seconds string)
  redis.call("ZREMRANGEBYSCORE",KEYS[1],0,ARGV[2])
  redis.call("ZADD",KEYS[1],ARGV[1],ARGV[1])
  local count = redis.call("ZCARD",KEYS[1])
  redis.call("EXPIRE",KEYS[1],ARGV[3])
  return count

## LockService — src/services/lock.service.ts

acquire(key: string, ttl: number): Promise<string | null>
  SET key uuid NX PX ttl — return uuid token on success, null on fail

release(key: string, token: string): Promise<boolean>
  run RELEASE_LOCK_SCRIPT

withLock<T>(key, ttl, fn, retries=3, backoff=50): Promise<T>
  Loop retries times:
    token = await acquire
    if token: try { return await fn() } finally { await release(key, token) }
    else: await sleep(backoff * 2^attempt)
  throw AppError LOCK_UNAVAILABLE 503

## SlotService — src/modules/slot/slot.service.ts

initSlotAvailability(slotId, available):
  SET slot:{slotId}:available {available}
  No TTL — key persists until slot is deactivated

getSlots(providerId, date, requestingUserId):
1. cacheKey = CACHE_KEYS.providerSlots(providerId, date)
2. cached = await cacheService.get(cacheKey)
3. On miss:
   a. Query Mongo: Slot.find({ providerId, date: { $gte: startOfDay, $lte: endOfDay }, isActive:true })
   b. For slots missing Redis availability key (GET returns null): call initSlotAvailability
   c. For each slot: queueLength = await redis.zcard(`queue:${slot.id}`)
   d. Build SlotResponseDTO[] — see shape below
   e. Cache slot list (base data only, no userBookingStatus) for CACHE_TTL_SECONDS
4. Attach userBookingStatus per request:
   Query Booking: { userId: requestingUserId, slotId: { $in: slotIds }, status: { $in: [pending,confirmed] } }
   Query QueueEntry: { userId: requestingUserId, slotId: { $in: slotIds }, status: { $in: [waiting,notified] } }
   Map onto each slot
5. Return enriched array

SlotResponseDTO shape:
{
  id, providerId, startTime, endTime, date,
  capacity, booked, available, isFull,
  status: "available" | "full" | "closed",
  queueLength,
  canBook: status === "available",
  canJoinQueue: status === "full",
  userBookingStatus: null | "pending" | "booked" | "in_queue" | "notified"
}

getSlotById(slotId, requestingUserId): same enrichment for single slot, check Redis availability, attach userBookingStatus.

generateSlots(providerId, requestingUserId, startDate, endDate):
  Validate range <= SLOT_GENERATE_MAX_DAYS — throw VALIDATION_ERROR if exceeded
  Find provider, verify provider.userId === requestingUserId — throw FORBIDDEN if not
  slots = generateSlots(provider, startDate, endDate) utility
  result = await Slot.insertMany(slots, { ordered: false }) — catch E11000 (duplicates skip silently)
  For each inserted slot: await initSlotAvailability(slot.id, slot.capacity)
  Return { generated: result.length, total: slots.length }

## BookingService — src/modules/booking/booking.service.ts

bookSlot(slotId: string, userId: string):
1. Parallel check:
   existing = Booking.findOne({ slotId, userId, status: { $in: [pending, confirmed] } })
   inQueue = QueueEntry.findOne({ slotId, userId, status: { $in: [waiting, notified] } })
   Await both — throw ALREADY_BOOKED or ALREADY_IN_QUEUE if found
2. Run BOOK_SLOT_SCRIPT on slot:{slotId}:available:
   result = await redis.eval(BOOK_SLOT_SCRIPT, 1, `slot:${slotId}:available`)
   if result === -2:
     slot = await Slot.findById(slotId) — throw NOT_FOUND if null
     await initSlotAvailability(slotId, slot.capacity - slot.booked)
     result = await redis.eval(BOOK_SLOT_SCRIPT, 1, `slot:${slotId}:available`)
   if result === -1: throw AppError SLOT_FULL 409 with { canJoinQueue: true, slotId }
3. Start Mongoose session:
   try {
     await session.withTransaction(async () => {
       await Slot.findByIdAndUpdate(slotId, { $inc: { booked: 1 } }, { session })
       booking = await Booking.create([{ slotId, userId, status: pending,
         expiresAt: new Date(Date.now() + BOOKING_EXPIRY_MINUTES * 60000) }], { session })
     })
   } catch (err) {
     await redis.eval(RELEASE_SLOT_SCRIPT, 1, `slot:${slotId}:available`)  // rollback Redis
     throw err
   }
4. Enqueue expiration job: jobId=expire:booking:{bookingId}, delay=BOOKING_EXPIRY_MINUTES*60000
5. Invalidate: CACHE_KEYS.slotAvailability(slotId), CACHE_KEYS.providerSlots pattern
6. Emit booking:confirmed WebSocket to userId
7. Return { booking, slot: await slotService.getSlotById(slotId, userId) }

confirmBooking(bookingId, userId):
  Find booking — throw NOT_FOUND if null, FORBIDDEN if userId mismatch, VALIDATION_ERROR if status !== pending
  Update status confirmed, confirmedAt now
  Remove BullMQ job: await expirationQueue.remove(`expire:booking:${bookingId}`)
  Return updated booking

cancelBooking(bookingId, userId):
1. Find booking, verify ownership, verify status in [confirmed, pending] — throw appropriate errors
2. session.withTransaction:
   Update booking status cancelled
   Slot $inc booked -1
3. RELEASE_SLOT_SCRIPT on slot:{slotId}:available
4. Invalidate caches
5. Enqueue promotion job: { slotId, availableSeats: 1 }, jobId=promote:{slotId}
6. Emit booking:cancelled WebSocket to userId
7. Return updated booking

getMyBookings(userId, query):
  Booking.find({ userId }).populate(slotId).sort(-createdAt).paginate(query)

getAllBookings(query):
  Admin only. Filter by status, slotId, providerId. Paginate.

## QueueService — src/modules/queue/queue.service.ts

Queue key: queue:{slotId} — Sorted Set, score = Date.now() (FIFO pure)

joinQueue(slotId, userId):
1. slot = await Slot.findById(slotId) — throw NOT_FOUND if null
2. If slot.status !== "full" ? throw SLOT_NOT_FULL 409 (direct user to book instead)
3. Parallel: check active booking + existing queue entry — throw ALREADY_BOOKED or ALREADY_IN_QUEUE
4. score = Date.now()
5. ZADD queue:{slotId} score userId
6. position = ZRANK queue:{slotId} userId + 1
7. estimatedWait = calculateWaitTime(position, provider.slotDuration)
8. Create QueueEntry: { slotId, userId, status: waiting, estimatedWait }
9. INCR queue:length:{slotId}
10. Emit queue:joined WebSocket { position, estimatedWait }
11. Return { position, estimatedWait, queueLength: position }

promoteBatch(slotId, availableSeats):
  lockService.withLock(`lock:queue:${slotId}`, LOCK_TTL_MS, async () => {
    promoted = 0
    notifiedUsers = []
    while promoted < availableSeats:
      result = await redis.zpopmin(`queue:${slotId}`, 1)
      if empty ? break
      userId = result[0]
      entry = QueueEntry.findOne({ slotId, userId, status: waiting })
      if !entry ? continue (already handled, try next)
      deadline = new Date(Date.now() + QUEUE_CONFIRM_MINUTES * 60000)
      Update entry: status notified, notifiedAt now, confirmationDeadline deadline
      Enqueue queue-confirmation-expiry job:
        jobId=queue:expire:{slotId}:{userId}, delay=QUEUE_CONFIRM_MINUTES*60000
      Emit queue:your_turn WebSocket to userId: { slotId, confirmationDeadline: deadline }
      Enqueue notification: { type: queue_promoted, userId, email, payload: { slotId, deadline } }
      promoted++
      notifiedUsers.push(userId)
    await redis.decrby(`queue:length:${slotId}`, promoted)
    return { promoted, notifiedUsers }
  })

confirmQueueSlot(slotId, userId):
1. entry = QueueEntry.findOne({ slotId, userId, status: notified })
   if !entry ? throw NOT_FOUND
   if entry.confirmationDeadline < now ? throw QUEUE_CONFIRMATION_EXPIRED 410
2. lockService.withLock(`lock:slot:${slotId}`, LOCK_TTL_MS, async () => {
   result = await redis.eval(BOOK_SLOT_SCRIPT, 1, `slot:${slotId}:available`)
   if result === -1 ? throw SLOT_FULL 409 (edge case: call promoteBatch for next user)
   session.withTransaction:
     Slot $inc booked 1
     booking = Booking.create([{ slotId, userId, status: confirmed, confirmedAt: now,
       expiresAt: new Date(Date.now() + 365*24*60*60*1000) }])  // far future, confirmed no expiry
     QueueEntry update status confirmed
   on txn failure: RELEASE_SLOT_SCRIPT rollback
3. Invalidate caches
4. Enqueue notification: booking_confirmed
5. Emit queue:confirmed WebSocket to userId
6. Return { booking }

getPosition(slotId, userId):
  rank = await redis.zrank(`queue:${slotId}`, userId)
  if rank === null ? return null (not in queue)
  position = rank + 1
  entry = QueueEntry.findOne({ slotId, userId })
  Return { position, estimatedWait: entry.estimatedWait, status: entry.status,
           confirmationDeadline: entry.confirmationDeadline ?? null }

leaveQueue(slotId, userId):
  ZREM queue:{slotId} userId
  Update QueueEntry status left
  DECR queue:length:{slotId}
  cacheService.del(CACHE_KEYS.queueLength(slotId))
  Emit queue:left WebSocket to userId
  Return { success: true }

### promotion.worker.ts
1. slot = Slot.findById(slotId) — if not found ? return
2. available = slot.capacity - slot.booked

4. await queueService.promoteBatch(slotId, available)

1. entry = QueueEntry.findOne({ slotId, userId, status: notified }) — if not found or wrong status ? return (idempotent)
2. Update entry status expired
3. Enqueue BOOKING_PROMOTION: { slotId, availableSeats: 1 }
4. Enqueue NOTIFICATIONS: { type: queue_confirmation_expired, userId, email, payload }

### notification.worker.ts

### rate-limit.ts
Per request:
  key = `rl:${req.ip}:${req.path}`
  now = Date.now().toString()
  windowStart = (Date.now() - RATE_LIMIT_WINDOW_MS).toString()
  ttl = Math.ceil(RATE_LIMIT_WINDOW_MS / 1000).toString()
  count = await redis.eval(RATE_LIMIT_SCRIPT, 1, key, now, windowStart, ttl)
  if count > RATE_LIMIT_MAX:
    res.set('Retry-After', Math.ceil(RATE_LIMIT_WINDOW_MS / 1000).toString())
    throw AppError RATE_LIMIT_EXCEEDED 429

### idempotency.ts
Apply only to POST /bookings, POST /queue/:slotId/join, POST /queue/:slotId/confirm.
1. key = req.headers['idempotency-key'] — if not present ? next()
2. cacheKey = `idem:${req.user.userId}:${key}`
3. cached = await redis.get(cacheKey) — if found ? res.set('X-Idempotent-Replay','true').json(JSON.parse(cached)) and return
4. res.locals.idempotencyKey = cacheKey — controllers call storeIdempotentResponse after success
5. Export storeIdempotentResponse(cacheKey, data): redis.set(cacheKey, JSON.stringify(data), 'EX', IDEMPOTENCY_TTL_SECONDS)

## TokenService — src/modules/auth/token.service.ts

generateAccessToken(payload: { userId, role }):
  jti = uuid v4
  sign { userId, role, jti } with JWT_ACCESS_SECRET, expiresIn JWT_ACCESS_EXPIRES_IN
  return token string

generateRefreshToken(userId):
  jti = uuid v4
  token = sign { userId, jti } with JWT_REFRESH_SECRET, expiresIn JWT_REFRESH_EXPIRES_IN
  hash = SHA-256(token)
  ttl = parse JWT_REFRESH_EXPIRES_IN to seconds
  redis.set(`rt:${userId}:${jti}`, hash, 'EX', ttl)
  return token string

rotateRefreshToken(oldToken):
  decoded = jwt.verify(oldToken, JWT_REFRESH_SECRET) — throw INVALID_TOKEN on fail, TOKEN_EXPIRED on expiry
  { userId, jti } = decoded
  stored = await redis.get(`rt:${userId}:${jti}`)
  if !stored:
    await revokeAllUserTokens(userId)
    throw AppError REFRESH_TOKEN_REUSE 401
  if SHA-256(oldToken) !== stored ? same as above
  await redis.del(`rt:${userId}:${jti}`)
  accessToken = generateAccessToken({ userId, role: decoded.role })
  refreshToken = generateRefreshToken(userId)
  return { accessToken, refreshToken }

revokeAllUserTokens(userId):
  keys = await scanKeys(`rt:${userId}:*`)
  if keys.length ? await redis.del(...keys)

  user = await User.findOne({ email }).select('+password') — throw UNAUTHORIZED if not found
  valid = await user.comparePassword(password) — throw UNAUTHORIZED if false
  tokens = { accessToken, refreshToken }

## WebSocket — src/services/websocket.service.ts

import { Server } from 'socket.io'
import { createAdapter } from '@socket.io/redis-adapter'

initWebSocket(httpServer):
  io = new Server(httpServer, { cors: corsOptions, transports: ['websocket', 'polling'] })
  pubClient = redis.duplicate()
  subClient = redis.duplicate()
  io.adapter(createAdapter(pubClient, subClient))
  io.use((socket, next) => {
    token = socket.handshake.auth.token
    if !token ? return next(new Error('UNAUTHORIZED'))
    try { payload = jwt.verify(token, JWT_ACCESS_SECRET); socket.data.userId = payload.userId; next() }
    catch { next(new Error('INVALID_TOKEN')) }
  })
  io.on('connection', (socket) => {
    socket.join(`user:${socket.data.userId}`)
    socket.on('disconnect', () => logger.debug('WS disconnected', socket.id))
  })
  return io

emitToUser(userId, event, data):
  io.to(`user:${userId}`).emit(event, data)

async function start():
  1. env = parseEnv() — exits on failure

  3. await connectDatabase() with retry loop: 5 attempts, 5s between, log each attempt
  4. await connectRedis()
  5. httpServer = createServer(app)

       new QueueConfirmationExpiryWorker(redis),
       new NotificationWorker(redis),
     ]
  9. logger.info('All workers initialized')

const shutdown = async (signal: string) => {
  logger.info(`${signal} received`)
  httpServer.close(async () => {
    await Promise.all(workers.map(w => w.close()))
    await mongoose.disconnect()
    await redis.quit()
    logger.info('Graceful shutdown complete')
    process.exit(0)
  })
  setTimeout(() => { logger.error('Force exit'); process.exit(1) }, 30000)
}

process.on('SIGTERM', () => shutdown('SIGTERM'))
process.on('SIGINT',  () => shutdown('SIGINT'))
process.on('unhandledRejection', (reason) => { logger.error({ reason }, 'Unhandled rejection'); shutdown('unhandledRejection') })

start().catch(err => { logger.error(err, 'Startup failed'); process.exit(1) })

## Engineering Rules — non-negotiable

1. asyncHandler wraps every route handler — zero unhandled promise rejections
2. BOOK_SLOT_SCRIPT is the sole overbooking prevention for direct booking — no locks needed
3. On any failure after BOOK_SLOT_SCRIPT succeeds ? RELEASE_SLOT_SCRIPT immediately (rollback)
4. LockService used only for promoteBatch — the one multi-step queue operation
5. Every worker checks entity state before acting — fully idempotent
6. Cache invalidated on every write — never serve stale slot availability
7. Unique index { slotId, userId } on bookings — DB-level last line of defense
8. Refresh token reuse detected ? revoke all tokens for that user
9. Rate limiter key = IP + path — different limits possible per route
10. Env parse at startup — server never starts with missing config
11. Pino redacts authorization header, password, token in all log output
12. WebSocket unauthenticated connections rejected at handshake
13. POST /bookings ? 409 SLOT_FULL with canJoinQueue:true — never auto-enqueues
14. POST /queue/:id/join ? 409 SLOT_NOT_FULL if available seats exist — direct to book
15. promoteBatch handles capacity > 1 — promotes multiple users atomically
16. Idempotency middleware on booking + queue join + queue confirm — safe for client retries
17. Socket.io uses Redis adapter — horizontal scaling ready
18. FIFO queue via timestamp score — simple, fair, no starvation possible
